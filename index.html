Database Setup

First, create the database ‚Äî like a filing cabinet holding all your tables (folders) with data (files).

‚ö†Ô∏è

Before You Begin

Ensure SQL Server Management Studio (SSMS) is installed and connected.

  1. Switch to Master Database

    Start in "master" ‚Äî the control room for creating databases.

    SQL
    USE master;
GO

  2. Remove Old Database (If Exists)

    Delete any old version to start fresh.

    SQL
    IF EXISTS (SELECT name FROM sys.databases WHERE name = N'DB_LastNameFirstName01')
BEGIN
    ALTER DATABASE [DB_LastNameFirstName01] SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
    DROP DATABASE [DB_LastNameFirstName01];
END
GO

  3. Create New Database

    Create and switch to your new database.

    SQL
    CREATE DATABASE [DB_LastNameFirstName01];
GO
USE [DB_LastNameFirstName01];
GO
    üí°

    Tip

    Replace "LastNameFirstName" with your name: DB_SmithJohn01

Create 8 tables in order ‚Äî parent tables first, then child tables with foreign keys.

Order Matters!

Create parent tables (Categories) before child tables (Products).

Product category groups

SQL
CREATE TABLE TBL_PRODUCT_CATEGORIES (
    CATEGORY_ID   BIGINT IDENTITY(1,1) NOT NULL,
    CATEGORY_NAME VARCHAR(100) NOT NULL,
    CONSTRAINT PK_PRODUCT_CATEGORIES PRIMARY KEY (CATEGORY_ID)
); GO

CREATE TABLE TBL_WAREHOUSES (

    CONSTRAINT PK_WAREHOUSES PRIMARY KEY (WAREHOUSE_ID)

Customer accounts

    CONSTRAINT PK_CUSTOMERS PRIMARY KEY (CUSTOMER_ID)

TBL_PRODUCTS

Items for sale

SQL
CREATE TABLE TBL_PRODUCTS (
    PRODUCT_ID    BIGINT IDENTITY(1,1) NOT NULL,
    PRODUCT_NAME  VARCHAR(200) NOT NULL,
    DESCRIPTION   VARCHAR(500) NULL,
    STANDARD_COST DECIMAL(5,2) NOT NULL,
    LIST_PRICE    DECIMAL(5,2) NOT NULL,
    CATEGORY_ID   BIGINT NOT NULL,
    CONSTRAINT PK_PRODUCTS PRIMARY KEY (PRODUCT_ID),
    CONSTRAINT FK_PRODUCTS_CATEGORY FOREIGN KEY (CATEGORY_ID) REFERENCES TBL_PRODUCT_CATEGORIES(CATEGORY_ID)
); GO

TBL_ORDERS

Purchase orders

SQL
CREATE TABLE TBL_ORDERS (
    ORDER_ID    BIGINT IDENTITY(1,1) NOT NULL,
    CUSTOMER_ID BIGINT NOT NULL,
    STATUS      VARCHAR(50) NOT NULL,
    EMPLOYEE_ID BIGINT NOT NULL,
    ORDER_DATE  VARCHAR(20) NOT NULL,
    CONSTRAINT PK_ORDERS PRIMARY KEY (ORDER_ID),
    CONSTRAINT FK_ORDERS_CUSTOMER FOREIGN KEY (CUSTOMER_ID) REFERENCES TBL_CUSTOMERS(CUSTOMER_ID),
    CONSTRAINT FK_ORDERS_EMPLOYEE FOREIGN KEY (EMPLOYEE_ID) REFERENCES TBL_EMPLOYEES(EMPLOYEE_ID)
); GO

TBL_ITEMS

Order line items

SQL
CREATE TABLE TBL_ITEMS (
    ITEM_ID    BIGINT IDENTITY(1,1) NOT NULL,
    ORDER_ID   BIGINT NOT NULL,
    PRODUCT_ID BIGINT NOT NULL,
    QUANTITY   INT NOT NULL,
    PRICE      DECIMAL(5,2) NOT NULL,
    CONSTRAINT PK_ITEMS PRIMARY KEY (ITEM_ID),
    CONSTRAINT FK_ITEMS_ORDER FOREIGN KEY (ORDER_ID) REFERENCES TBL_ORDERS(ORDER_ID),
    CONSTRAINT FK_ITEMS_PRODUCT FOREIGN KEY (PRODUCT_ID) REFERENCES TBL_PRODUCTS(PRODUCT_ID)
); GO

TBL_INVENTORIES

Stock levels (composite key)

üìå

Composite Key

Two columns form the primary key together.

SQL
CREATE TABLE TBL_INVENTORIES (
    PRODUCT_ID   BIGINT NOT NULL,
    WAREHOUSE_ID BIGINT NOT NULL,
    QUANTITY     INT NOT NULL DEFAULT 0,
    CONSTRAINT PK_INVENTORIES PRIMARY KEY (PRODUCT_ID, WAREHOUSE_ID),
    CONSTRAINT FK_INV_PRODUCT FOREIGN KEY (PRODUCT_ID) REFERENCES TBL_PRODUCTS(PRODUCT_ID),
    CONSTRAINT FK_INV_WAREHOUSE FOREIGN KEY (WAREHOUSE_ID) REFERENCES TBL_WAREHOUSES(WAREHOUSE_ID)
); GO

Complete Script

Copy the entire script to run everything at once.

Full Script
USE master; GO
IF EXISTS(SELECT name FROM sys.databases WHERE name=N'DB_LastNameFirstName01')
BEGIN ALTER DATABASE [DB_LastNameFirstName01] SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
DROP DATABASE [DB_LastNameFirstName01]; END GO

CREATE TABLE TBL_PRODUCT_CATEGORIES(CATEGORY_ID BIGINT IDENTITY(1,1) NOT NULL,CATEGORY_NAME VARCHAR(100) NOT NULL,CONSTRAINT PK_PRODUCT_CATEGORIES PRIMARY KEY(CATEGORY_ID)); GO
CREATE TABLE TBL_WAREHOUSES(WAREHOUSE_ID BIGINT IDENTITY(1,1) NOT NULL,WAREHOUSE_NAME VARCHAR(100) NOT NULL,CONSTRAINT PK_WAREHOUSES PRIMARY KEY(WAREHOUSE_ID)); GO
CREATE TABLE TBL_CUSTOMERS(CUSTOMER_ID BIGINT IDENTITY(1,1) NOT NULL,NAME VARCHAR(150) NOT NULL,ADDRESS VARCHAR(255) NULL,CREDIT_LIMIT INT DEFAULT 0,CONSTRAINT PK_CUSTOMERS PRIMARY KEY(CUSTOMER_ID)); GO
CREATE TABLE TBL_EMPLOYEES(EMPLOYEE_ID BIGINT IDENTITY(1,1) NOT NULL,FIRST_NAME VARCHAR(100) NOT NULL,LAST_NAME VARCHAR(100) NOT NULL,EMAIL VARCHAR(150) NOT NULL,PHONE INT NULL,HIRE_DATE VARCHAR(20) NOT NULL,JOB_TITLE VARCHAR(100) NOT NULL,CONSTRAINT PK_EMPLOYEES PRIMARY KEY(EMPLOYEE_ID)); GO
CREATE TABLE TBL_PRODUCTS(PRODUCT_ID BIGINT IDENTITY(1,1) NOT NULL,PRODUCT_NAME VARCHAR(200) NOT NULL,DESCRIPTION VARCHAR(500) NULL,STANDARD_COST DECIMAL(5,2) NOT NULL,LIST_PRICE DECIMAL(5,2) NOT NULL,CATEGORY_ID BIGINT NOT NULL,CONSTRAINT PK_PRODUCTS PRIMARY KEY(PRODUCT_ID),CONSTRAINT FK_PRODUCTS_CATEGORY FOREIGN KEY(CATEGORY_ID) REFERENCES TBL_PRODUCT_CATEGORIES(CATEGORY_ID)); GO
CREATE TABLE TBL_ORDERS(ORDER_ID BIGINT IDENTITY(1,1) NOT NULL,CUSTOMER_ID BIGINT NOT NULL,STATUS VARCHAR(50) NOT NULL,EMPLOYEE_ID BIGINT NOT NULL,ORDER_DATE VARCHAR(20) NOT NULL,CONSTRAINT PK_ORDERS PRIMARY KEY(ORDER_ID),CONSTRAINT FK_ORDERS_CUSTOMER FOREIGN KEY(CUSTOMER_ID) REFERENCES TBL_CUSTOMERS(CUSTOMER_ID),CONSTRAINT FK_ORDERS_EMPLOYEE FOREIGN KEY(EMPLOYEE_ID) REFERENCES TBL_EMPLOYEES(EMPLOYEE_ID)); GO
CREATE TABLE TBL_ITEMS(ITEM_ID BIGINT IDENTITY(1,1) NOT NULL,ORDER_ID BIGINT NOT NULL,PRODUCT_ID BIGINT NOT NULL,QUANTITY INT NOT NULL,PRICE DECIMAL(5,2) NOT NULL,CONSTRAINT PK_ITEMS PRIMARY KEY(ITEM_ID),CONSTRAINT FK_ITEMS_ORDER FOREIGN KEY(ORDER_ID) REFERENCES TBL_ORDERS(ORDER_ID),CONSTRAINT FK_ITEMS_PRODUCT FOREIGN KEY(PRODUCT_ID) REFERENCES TBL_PRODUCTS(PRODUCT_ID)); GO
CREATE TABLE TBL_INVENTORIES(PRODUCT_ID BIGINT NOT NULL,WAREHOUSE_ID BIGINT NOT NULL,QUANTITY INT NOT NULL DEFAULT 0,CONSTRAINT PK_INVENTORIES PRIMARY KEY(PRODUCT_ID,WAREHOUSE_ID),CONSTRAINT FK_INV_PRODUCT FOREIGN KEY(PRODUCT_ID) REFERENCES TBL_PRODUCTS(PRODUCT_ID),CONSTRAINT FK_INV_WAREHOUSE FOREIGN KEY(WAREHOUSE_ID) REFERENCES TBL_WAREHOUSES(WAREHOUSE_ID)); GO
